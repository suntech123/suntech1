## copy and paste in powershell window


$battery = Get-CimInstance -Namespace root/wmi -ClassName BatteryFullChargedCapacity
$design = Get-CimInstance -Namespace root/wmi -ClassName BatteryStaticData
$health = ($battery.FullChargedCapacity / $design.DesignedCapacity) * 100
Write-Host "Battery Health: $([math]::Round($health, 2))%"
